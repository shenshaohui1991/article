### 变量的解构赋值
+ 数组的解构赋值
```
	var [a,b,c] = [1,2,3];
	a // 1
	b // 2
	c // 3

	let [foo,[[bar],baz]] = [1,[[2],3]];
	foo // 1
	bar // 2
	baz // 3

	let [,,third] = ['1','2','3']
	third // 3

	let [x,,y] = [1,2,3];
	x // 1
	y // 3

	let [head, ...tail] = [1,2,3,4];
	head // 1
	tail // 2,3,4

	// 报错
	let [h1] = 1;
	let [h2] = false;
	let [h3] = NaN;
	let [h4] = undefined;
	let [h5] = null;
	等号右边必须是可以遍历的结构
```
