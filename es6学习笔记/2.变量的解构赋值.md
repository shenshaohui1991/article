### 变量的解构赋值
+ 数组的解构赋值
```
	var [a,b,c] = [1,2,3];
	a // 1
	b // 2
	c // 3

	let [foo,[[bar],baz]] = [1,[[2],3]];
	foo // 1
	bar // 2
	baz // 3

	let [,,third] = ['1','2','3']
	third // 3

	let [x,,y] = [1,2,3];
	x // 1
	y // 3

	let [head, ...tail] = [1,2,3,4];
	head // 1
	tail // 2,3,4

	// 报错
	let [h1] = 1;
	let [h2] = false;
	let [h3] = NaN;
	let [h4] = undefined;
	let [h5] = null;
	等号右边必须是可以遍历的结构
```
+ 对象的解构赋值
```
	对象解构和数组解构的区别是: 对象只需变量名和属性名对应即可，数组是按顺序对应
	var {foo, bar} = {foo: 'f', bar: 'b'};
	foo // 'f'
	bar // 'b'

	var {baz} = {bar: 'a'};
	baz // undefined

	// 对变量名和属性名不一致时
	var {foo: baz} = {foo: 'aaa'};
	baz // 'aaa'

	let obj = {first: 'hello', last: 'world'}
	let {first: f, last: l} = obj;
	f // 'hello'

	// 默认值 和数组解构相同  当 值 === undefined 取默认值 
	var {x = 3} = {x: undefined};
	x // 3

	var {x = 3} = {x: null};

	// 对于已经声明的变量用于解构赋值，需要注意语法
	var x;
	{x} = {x:1} // SyntaxError: syntax error  把{x}当成代码块
	({x} = {x:1})

	// 对象的方法的复制
	let {max,min,sin,cos} = Math;
```
+ 字符串的解构赋值
```
	const [a, b, c, d, e] = 'hello';
	a // 'h'
	b // 'e'
	c // 'l'
	d // 'l'
	e // 'o'

	let {length: len} = 'hello'
	len // 5
```
+ 函数参数的解构赋值
```
	// 类似对象解构赋值   默认值
	function({x,y} = {x = 0, y = 1}){
		return x + y;
	}

	move({x:3, y:10}); // 13
	move({}); // undefined
	move(); // 1
```
+ 解构赋值的作用
```
	// 交换变量值, 十分简洁
	[x, y] = [y, x];

	// 一次返回多个值
	function example(){
		return [1, 2, 3];
	}
	let [a, b, c] = example();

	// 函数参数的赋值
	function f([x,y,z]){
		...
	}
	f([1,3,5]);

	// 快速解析JSON对象
	var json = {
		id: 42,
		status: 1,
		data: [1,2,3]
	};
	let [id, status] = json;

	// 遍历MAP
	var map = new Map();
	map.set('first', 'hello');
	map.set('second', 'world');

	for(let [key, value] of map){
		console.log(key + ' ' + value);
	}

	// 加载模块 ,  console,print 方法名
	const {console,print} = require('haha');
```
