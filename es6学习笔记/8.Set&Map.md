### Set Map
+ Set

```

	// 创建
	var set = new Set();
	//  添加, 不会添加重复的值
	set.add(1);
	// set大小
	set.size; // 1
	set.add(2);
	// 删除
	set.delete(2); // true
	set.size; // 1
	set.add(3);
	// 判断是否存在
	set.has(3);// true
	// 清除
	set.clear(); // 没有返回值
	set.size; // 0

	// 小例子
	function divs(){
		return [...document.querySelectorAll('div')];
	}	

	var divs = new Set(divs());

	// array set 互相转换的函数
	Array.from(set);
	new Set(array);
	
	// 利用 Array.from 进行数组去重
	function uniq(array){
		return Array.from(new Set(array));
	}

	// 遍历操作
	// keys    返回所有键
	// values   返回所有键
	// entries  返回所有键：所有键
	// forEach 

	let set = new Set(['red', 'green', 'blue']);

	for ( let item of set.keys() ){
	  console.log(item);
	}
	// red
	// green
	// blue

	for ( let item of set.values() ){
	  console.log(item);
	}
	// red
	// green
	// blue

	for ( let item of set.entries() ){
	  console.log(item);
	}
	// ["red", "red"]
	// ["green", "green"]
	// ["blue", "blue"]
	
	// set本身也是可以遍历的

```

+ Map

```
		
	// 和对象的区别，对象是 字符串-值    MAP 是 值-值
	// 基本操作
	let map = new Map();

	// 设置
	map.set('i',1);
	map.set('j',2);
	map.set('obj',{x:1});
	
	// get 
	map.get('obj'); // {x:1}

	// size
	map.size // 3

	// has
	map.has('i'); // true

	// delete
	map.delete('j'); //true

	// clear
	map.clear();

	// map < = > 数组
	let map = new Map([
		[1, 'one'],
		[2, 'two'],
		[3, 'three']
	]);

	[...map.keys()]; // [1,2,3]
	[...map.values()]; // ['one','two','three']
	[...map.entries()]; // [[1,'one'],[2,'two'],[3,'three']]

```
